import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { Reserva } from '../interfaces/interface';

@Injectable({
  providedIn: 'root'
})
export class ReservaService {

  constructor(private http: HttpClient) { }

  /**
   * Obtiene la lista de usuarios.
   * @returns Un Observable que emite un array de objetos Reserva.
   */
  /* getUsuarios(): Observable<Reserva[]> { // no se usa???
    //esto esta mal?=?
    return this.http.get<Reserva[]>('http://localhost:3000/reservas');
  } */

  /**
   * Obtiene todas las reservas.
   * @returns Una promesa que se resuelve con la respuesta HTTP que contiene todas las reservas.
   */
  obtenerTodasReservas() {
    return this.http.get(`http://localhost:3000/reservas`);
  }

  /**
   * Realiza una reserva.
   * @param reserva - La reserva a realizar.
   * @returns Una promesa que se resuelve con la respuesta del servidor.
   */
  reservar(reserva: any) {
    // console.log('servicio' + reserva)
    return this.http.post('http://localhost:3000/reservas-insert', reserva);
  }

  /**
   * Obtiene las reservas de un usuario.
   * @param id_usuario - El ID del usuario.
   * @returns Una solicitud HTTP para obtener las reservas del usuario.
   */
  obtenerReservas(id_usuario: number) {
    return this.http.get(`http://localhost:3000/reservas-obtener-reservas?id_usuario=${id_usuario}`);
  }
 
  /**
   * Elimina una reserva seg√∫n su ID.
   * @param id_reserva - El ID de la reserva a eliminar.
   * @returns Una promesa que se resuelve cuando se completa la eliminaci√≥n de la reserva.
   */
  borrarReserva(id_reserva:number) {
    return this.http.delete(`http://localhost:3000/reservas-delete?id_reserva=${id_reserva}`);
  }
}
