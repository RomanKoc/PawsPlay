import { CommonModule } from '@angular/common';
import { Component } from '@angular/core';
import { DialogModule } from 'primeng/dialog';

@Component({
  selector: 'app-footer',
  standalone: true,
  imports: [CommonModule, DialogModule],
  templateUrl: './footer.component.html',

})
export class FooterComponent {

  /**
   * Indica si se deben mostrar las preguntas o no.
   * @default false
   */
  preguntas:  boolean = false;
  /**
   * Indica si deben mostar las política de privacidad.
   * @default false
   */
  privacidad: boolean = false;
  /**
   * Indica si deben mostrar los términos.
   * @default false
   */
  terminos:   boolean = false;

  /**
   * Muestan los dialog al pulsarse
   */
  showPreguntas() {
    this.preguntas  = true;
  }

  /**
   * Muestra la sección de privacidad.
   */
  showPrivacidad() {
    this.privacidad = true;
  }

  /**
   * Muestra los términos y condiciones.
   */
  showTerminos() {
    this.terminos   = true;
  }

}
